Java Collections Framework

The Collections Framework provides classes and interfaces for storing and manipulating groups of objects.

List

import java.util.*;

// ArrayList
List<String> list = new ArrayList<>();
list.add("Apple");
list.add("Banana");
list.get(0); // "Apple"

// LinkedList
List<Integer> linkedList = new LinkedList<>();

Set

// HashSet (no duplicates, unordered)
Set<String> set = new HashSet<>();
set.add("Red");
set.add("Blue");

// TreeSet (sorted)
Set<Integer> treeSet = new TreeSet<>();

Map

// HashMap (key-value pairs)
Map<String, Integer> map = new HashMap<>();
map.put("Alice", 25);
map.put("Bob", 30);
map.get("Alice"); // 25

// TreeMap (sorted keys)
Map<String, String> treeMap = new TreeMap<>();
